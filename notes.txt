- To see what pages it belongs in vscode
go to settings
check the controls whether...

-dynamic metadata
-%s will be overriden the individual title of each pages
export const metadata = {
  title: {
    template: "%s | The Wild Oasis",
    default: "Welcome to The Wild Oasis",
  }
};
